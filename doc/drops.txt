drops(3)
========

NAME
----
drops - start/stop drops file sharing service

SYNOPSIS
--------
----
//  Constructor, creates a new drops instance running on the specified
//  directory. The instance runs in the background until you destroy it.
CZMQ_EXPORT drops_t *
    drops_new (const char *path);

//  Destructor, ends and destroys a drops service.
CZMQ_EXPORT void
    drops_destroy (drops_t **self_p);

//  Self test of this class
CZMQ_EXPORT void
    drops_test (bool verbose);
----

DESCRIPTION
-----------

To be written.


EXAMPLE
-------
.From drops_test method
----
    //  Create temporary directory for test files
#   define TESTDIR ".test_drops"
    zsys_dir_create (TESTDIR);

    //  Create and destroy an drops instance
    drops_t *drops = drops_new (TESTDIR);
    drops_destroy (&drops);

    //  Delete all test files

    zdir_destroy (&dir);
